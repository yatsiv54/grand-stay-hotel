class MyBookingsPage extends StatefulWidget {
  const MyBookingsPage({super.key});

  @override
  State<MyBookingsPage> createState() => _MyBookingsPageState();
}

class _MyBookingsPageState extends State<MyBookingsPage> {
  BookingStatus _tab = BookingStatus.active;
  late final BookingCubit _cubit;
  late final EntertainmentRepository _entRepo;
  Map<String, EntertainmentItem> _entById = const {};

  @override
  void initState() {
    super.initState();
    _cubit = getIt<BookingCubit>()..loadAll();
    _entRepo = getIt<EntertainmentRepository>();
    _loadEntertainments();
  }

  @override
  Widget build(BuildContext context) {
    return BlocBuilder<BookingCubit, BookingState>(
      bloc: _cubit,
      builder: (context, state) {
        final theme = Theme.of(context);
        final items = state.tab == BookingStatus.active
            ? state.active
            : state.history;
        final grouped = _groupByCategory(items);

        return Container(
          color: AppTheme.surface,
          child: Column(
            children: [
              _TabSwitcher(
                tab: state.tab,
                onChanged: (value) {
                  _tab = value;
                  _cubit.setTab(value);
                },
              ),
              Expanded(
                child: ListView(
                  padding: const EdgeInsets.symmetric(
                    horizontal: 20,
                    vertical: 15,
                  ),
                  children: [
                    const SizedBox(height: 16),
                    if (state.loading)
                      const Center(child: CircularProgressIndicator())
                    else
                      ...grouped.entries.map(
                        (entry) => Padding(
                          padding: const EdgeInsets.only(bottom: 18),
                          child: Column(
                            crossAxisAlignment: CrossAxisAlignment.start,
                            children: [
                              Text(
                                entry.key,
                                style: theme.textTheme.titleMedium,
                              ),
                              const SizedBox(height: 8),
                              ...entry.value.map(
                                (booking) => Padding(
                                  padding: const EdgeInsets.only(bottom: 10),
                                  child: _BookingCard(
                                    entry: booking,
                                    isHistory:
                                        state.tab == BookingStatus.history ||
                                        booking.status == BookingStatus.history,
                                    onActionTap: () {
                                      if (booking.category ==
                                          BookingCategory.activities) {
                                        _openActivityDetail(context, booking);
                                      } else {
                                        _openQr(context, booking);
                                      }
                                    },
                                  ),
                                ),
                              ),
                            ],
                          ),
                        ),
                      ),
                  ],
                ),
              ),
            ],
          ),
        );
      },
    );
  }

  Map<String, List<BookingEntry>> _groupByCategory(List<BookingEntry> items) {
    final order = <BookingCategory>[
      BookingCategory.rooms,
      BookingCategory.deals,
      BookingCategory.activities,
      BookingCategory.dining,
    ];
    final labels = {
      BookingCategory.rooms: 'Rooms',
      BookingCategory.deals: 'Exclusive deals',
      BookingCategory.activities: 'Activities',
      BookingCategory.dining: 'Dining',
    };

    final Map<String, List<BookingEntry>> grouped = {};
    for (final cat in order) {
      final filtered = items.where((e) => e.category == cat).toList();
      if (filtered.isNotEmpty) {
        grouped[labels[cat]!] = filtered;
      }
    }
    return grouped;
  }

  void _openQr(BuildContext context, BookingEntry entry) {
    final info = _toQrInfo(entry);
    if (info == null) {
      ScaffoldMessenger.of(
        context,
      ).showSnackBar(const SnackBar(content: Text('QR data unavailable')));
      return;
    }
    context.push('/qr/show', extra: info);
  }

  Future<void> _openActivityDetail(
    BuildContext context,
    BookingEntry entry,
  ) async {
    if (_entById.isEmpty) {
      await _loadEntertainments();
    }
    final item = _findEntertainment(entry);
    if (item != null) {
      if (context.mounted) {
        context.push('/entertainment/detail', extra: item);
      }
      return;
    }
    if (context.mounted) {
      context.push('/entertainment');
    }
  }

  EntertainmentItem? _findEntertainment(BookingEntry entry) {
    final id = _activityId(entry.id);
    if (id != null && _entById.containsKey(id)) return _entById[id];
    for (final item in _entById.values) {
      if (item.title.trim().toLowerCase() == entry.title.trim().toLowerCase()) {
        return item;
      }
      if (entry.id.contains(item.id) || item.id.contains(entry.id)) {
        return item;
      }
      if (entry.assetPlaceholder == item.imageAsset) return item;
    }
    return null;
  }

  String? _activityId(String entryId) {
    if (!entryId.startsWith('ent_')) return null;
    final raw = entryId.substring(4);
    final lastUnderscore = raw.lastIndexOf('_');
    if (lastUnderscore <= 0) return raw;
    final candidate = raw.substring(0, lastUnderscore);
    return candidate.isNotEmpty ? candidate : raw;
  }

  QrInfo? _toQrInfo(BookingEntry entry) {
    final guestName =
        (entry.contactName != null && entry.contactName!.trim().isNotEmpty)
        ? entry.contactName!.trim()
        : 'Guest';
    final phone =
        (entry.contactPhone != null && entry.contactPhone!.trim().isNotEmpty)
        ? entry.contactPhone!.trim()
        : '-';

    switch (entry.category) {
      case BookingCategory.rooms:
        return RoomQrInfo(
          img: entry.assetPlaceholder,
          subtitle: entry.detailSecondary ?? entry.detailPrimary ?? '',
          name: guestName,
          phoneNumber: phone,
          placeName: entry.title,
          hint: entry.hint ?? _defaultHint(entry.category),
          roomNumber: entry.roomNumber?.trim().isNotEmpty == true
              ? entry.roomNumber!.trim()
              : '',
          quantity: _quantityFrom(entry.guests, entry.adults, entry.children),
          preferences: entry.preferences,
          checkInDate:
              entry.checkInDate ??
              entry.eventDate?.toIso8601String().split('T').first ??
              _extractDate(entry.detailPrimary) ??
              '-',
          checkOutDate:
              entry.checkOutDate ??
              _fallbackNextDay(entry) ??
              _extractDate(entry.detailSecondary) ??
              '-',
        );
      case BookingCategory.dining:
        return DiningQrInfo(
          img: entry.assetPlaceholder,
          subtitle: entry.detailSecondary ?? '',
          name: guestName,
          phoneNumber: phone,
          placeName: entry.title,
          hint: entry.hint ?? _defaultHint(entry.category),
          quantity: _quantityFrom(null, entry.adults, entry.children),
          table: entry.table?.trim().isNotEmpty == true
              ? entry.table!.trim()
              : '',
          date:
              entry.date ??
              entry.eventDate?.toIso8601String().split('T').first ??
              _extractDate(entry.detailPrimary) ??
              '-',
          time:
              entry.time ??
              _extractTime(entry.detailPrimary) ??
              _formatTime(entry.eventDate) ??
              '-',
        );
      case BookingCategory.deals:
      case BookingCategory.activities:
        final date =
            entry.date ??
            entry.eventDate?.toIso8601String().split('T').first ??
            _extractDate(entry.detailPrimary) ??
            '-';
        final time =
            entry.time ??
            _extractTime(entry.detailPrimary) ??
            _formatTime(entry.eventDate) ??
            '-';
        return OfferQrInfo(
          img: entry.assetPlaceholder,
          subtitle: entry.detailSecondary ?? entry.detailPrimary ?? '',
          name: guestName,
          phoneNumber: phone,
          placeName: entry.title,
          hint: entry.hint ?? _defaultHint(entry.category),
          date: date,
          time: time,
        );
    }
  }

  Future<void> _loadEntertainments() async {
    try {
      final items = await _entRepo.fetchItems();
      if (!mounted) return;
      setState(() {
        _entById = {for (final e in items) e.id: e};
      });
    } catch (_) {
      // ignore load errors; fallback is opening list page
    }
  }

  List<String> _quantityFrom(String? guests, int? adults, int? children) {
    final parts = <String>[];
    if (adults != null) parts.add('Adults: $adults');
    if (children != null) parts.add('Children: $children');
    if (parts.isNotEmpty) return parts;
    if (guests == null || guests.trim().isEmpty) return const [];
    return guests
        .split(RegExp(r'\s{2,}'))
        .map((e) => e.trim())
        .where((e) => e.isNotEmpty)
        .toList();
  }

  String? _extractDate(String? label) {
    if (label == null) return null;
    final parts = label.split(RegExp(r':\s*'));
    if (parts.length > 1) {
      final value = parts.last.trim();
      if (RegExp(r'^\d{4}-\d{2}-\d{2}').hasMatch(value)) return value;
    }
    final match = RegExp(r'(\d{4}-\d{2}-\d{2})').firstMatch(label);
    return match?.group(1);
  }

  String? _extractTime(String? label) {
    if (label == null) return null;
    final match = RegExp(
      r'(\d{1,2}:\d{2}(?:\s?(?:AM|PM))?)',
      caseSensitive: false,
    ).firstMatch(label);
    return match?.group(1);
  }

  String? _formatTime(DateTime? dt) {
    if (dt == null) return null;
    final h = dt.hour.toString().padLeft(2, '0');
    final m = dt.minute.toString().padLeft(2, '0');
    return '$h:$m';
  }

  String _fallbackNextDay(BookingEntry entry) {
    final checkIn = _extractDate(entry.detailPrimary);
    if (checkIn != null) {
      final dt = DateTime.tryParse(checkIn);
      if (dt != null) {
        return dt
            .add(const Duration(days: 1))
            .toIso8601String()
            .split('T')
            .first;
      }
    }
    return _extractDate(entry.detailSecondary) ??
        entry.eventDate?.toIso8601String().split('T').first ??
        '-';
  }

  String _defaultHint(BookingCategory category) {
    switch (category) {
      case BookingCategory.rooms:
        return 'Show this code at the reception during check-in';
      case BookingCategory.dining:
        return 'Scan this code at the entrance for quicker check-in';
      case BookingCategory.deals:
      case BookingCategory.activities:
        return 'Show this code at the entrance during the event';
    }
  }
}

class _TabSwitcher extends StatelessWidget {
  const _TabSwitcher({required this.tab, required this.onChanged});

  final BookingStatus tab;
  final ValueChanged<BookingStatus> onChanged;

  @override
  Widget build(BuildContext context) {
    return Container(
      height: 56,
      width: double.infinity,
      decoration: BoxDecoration(
        borderRadius: BorderRadius.circular(0),

          colors: [
            Color.fromRGBO(230, 33, 45, 1),
            Color.fromRGBO(232, 34, 42, 1),

          end: Alignment.bottomCenter,
        ),
        boxShadow: [
          BoxShadow(
            color: Colors.black.withOpacity(0.12),
            blurRadius: 10,
            offset: const Offset(0, 6),
          ),
        ],
      ),
      padding: const EdgeInsets.symmetric(horizontal: 8, vertical: 6),
      child: Row(
        children: [
          _TabButton(
            label: 'Active',
            icon: 'assets/images/bell.png',
            isActive: tab == BookingStatus.active,
            onTap: () => onChanged(BookingStatus.active),
          ),
          const SizedBox(width: 8),
          _TabButton(
            label: 'History',
            icon: 'assets/images/history.png',
            isActive: tab == BookingStatus.history,
            onTap: () => onChanged(BookingStatus.history),
          ),
        ],
      ),
    );
  }
}

class _BookingCard extends StatelessWidget {
  const _BookingCard({
    required this.entry,
    required this.onActionTap,
    this.isHistory = false,
  });

  final BookingEntry entry;
  final VoidCallback onActionTap;
  final bool isHistory;

  @override
  Widget build(BuildContext context) {
    final theme = Theme.of(context);
    final isEntertainment = entry.category == BookingCategory.activities;
    final bgColor = isHistory ? Colors.grey.shade200 : Colors.white;
    final shadow = isHistory
        ? null
        : [
            BoxShadow(
              color: Colors.black.withOpacity(0.06),
              blurRadius: 10,
              offset: const Offset(0, 4),
            ),
          ];

    return DecoratedBox(
      decoration: BoxDecoration(
        color: bgColor,
        borderRadius: BorderRadius.circular(14),
        boxShadow: shadow,
      ),
      child: Row(
        children: [
          _ImagePlaceholder(token: entry.assetPlaceholder),
          const SizedBox(width: 10),
          Expanded(
            child: Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                Text(
                  entry.title,
                  style: theme.textTheme.titleSmall?.copyWith(
                    color: AppTheme.textPrimary,
                  ),
                  maxLines: 2,
                ),
                const SizedBox(height: 6),
                if (entry.detailPrimary != null)
                  _RichDetail(text: entry.detailPrimary!),
                if (entry.detailSecondary != null)
                  _RichDetail(text: entry.detailSecondary!),
                // guests text hidden for dining per request
              ],
            ),
          ),
          const SizedBox(width: 8),
          if (!isHistory)
            Container(
              height: 96,
              alignment: Alignment.bottomLeft,
              padding: const EdgeInsets.fromLTRB(0, 0, 12, 10),
              child: _CtaButton(
                cta: isEntertainment ? BookingCta.arrow : entry.cta,
                onTap: onActionTap,
              ),
            ),
        ],
      ),
    );
  }
}

  const _ImagePlaceholder({required this.token});

  final String token;

class _RichDetail extends StatelessWidget {
  const _RichDetail({required this.text});
  final String text;

  @override
  Widget build(BuildContext context) {
    final theme = Theme.of(context);
    final parts = text.split(':');
    if (parts.length < 2) {
      return Text(
        text,
        style: theme.textTheme.bodyMedium?.copyWith(color: Colors.black),
      );
    }
    final label = parts.first;
    final value = text.substring(text.indexOf(':') + 1).trim();
    return RichText(
      text: TextSpan(
        style: theme.textTheme.bodyMedium?.copyWith(color: Colors.black),
        children: [
          TextSpan(
            text: '$label: ',
            style: theme.textTheme.bodyMedium?.copyWith(
              color: Colors.grey.shade700,
              fontWeight: FontWeight.w600,
            ),
          ),
          TextSpan(
            text: value,
            style: theme.textTheme.bodyMedium?.copyWith(
              color: Colors.black,
              fontWeight: FontWeight.w600,
            ),
          ),
        ],
      ),
    );
  }
}}

class _Fallback extends StatelessWidget {
  const _Fallback({required this.colors});
  final List<Color> colors;

  @override
  Widget build(BuildContext context) {
    return Container(
      decoration: BoxDecoration(
        gradient: LinearGradient(
          colors: colors,
          begin: Alignment.topLeft,
          end: Alignment.bottomRight,
        ),
      ),
      child: Icon(Icons.photo, color: Colors.white.withOpacity(0.85), size: 26),
    );
  }
}

List<Color> _gradientFor(String key) {
  switch (key) {
    case 'rooms_1':
      return const [Color(0xFFf1c18a), Color(0xFFd58858)];
    case 'deal_spa':
      return const [Color(0xFFf7a1b2), Color(0xFFd26983)];
    case 'deal_bike':
      return const [Color(0xFFffcf6f), Color(0xFFe09542)];
    case 'activity_jazz':
      return const [Color(0xFFd12b3d), Color(0xFF7b1e39)];
    case 'dining_sushi':
      return const [Color(0xFFf0a25e), Color(0xFFc25133)];
    default:
      return const [Color(0xFFd72f32), Color(0xFFb30f14)];
  }
}

            padding: const EdgeInsets.symmetric(horizontal: 2, vertical: 2),

              color: Colors.white,
              size: isQr ? 20 : 16,
            ),
          ),
        ),
      ),
    );
  }
}
